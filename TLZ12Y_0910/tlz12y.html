  1. git init - uj repo
  2. git clone - tavoli repo klonozasa helyi gepre
  3. git status - aktualis repo allapot megnezese

  5. git commit -m - commit uzenet
  6. git branch - meglevo branchek
  7. git branch - uj ag letrehozas
  8. git checkout - agvaltas
